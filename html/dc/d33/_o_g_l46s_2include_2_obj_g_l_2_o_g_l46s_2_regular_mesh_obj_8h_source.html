120  void cacheVertexLst(const ObjGL::RegularMesh &regMesh)
121  {
122 
123  glm::vec3 normal(0, 1, 0);
124 
125  for (int y = 0; y < _rows; y++)
126  {
127  for (int x = 0; x < _cols; x++)
128  {
129  glm::vec3 v(x, _heightMap.at(y * _rows + _cols), y);
130  glm::vec2 tC = glm::vec2(float(x) / float(_cols), float(y) / float(_rows));
131 
132  _vLst.add(ObjGL::Vertex(v, normal, tC));
133  }
134  }
135 
136  if (!_idVBO)
137  glGenBuffers(1, &_idVBO);
138 
139  glBindBuffer(GL_ARRAY_BUFFER, _idVBO);
140  glBufferData(GL_ARRAY_BUFFER, sizeof(ObjGL::Vertex)* _vLst.numVertices(), _vLst.ptrMem(), GL_STATIC_DRAW);
141 
142  glEnableClientState(GL_VERTEX_ARRAY);
143  glVertexPointer(3, GL_FLOAT, sizeof(ObjGL::Vertex), 0);
144 
145  glEnableClientState(GL_NORMAL_ARRAY);
146  glNormalPointer(GL_FLOAT, sizeof(ObjGL::Vertex), (void*)(sizeof(glm::vec3)));
147 
148  glEnableClientState(GL_TEXTURE_COORD_ARRAY);
149  glTexCoordPointer(2, GL_FLOAT, sizeof(ObjGL::Vertex), (void*)(sizeof(glm::vec3) * 2));
150 
151  glBindBuffer(GL_ARRAY_BUFFER, 0);
152 
153  _nV = (GLsizei)_vLst.numVertices();
154  _renMode = regMesh._renMode;
155 
156  bound(_vLst.minVertex(), _vLst.maxVertex());
157  }
